25  import org.codehaus.mojo.license.header.FileHeader;
26  
27  /**
28   * Contract to transform a file header to {@link FileHeader} in both way :
29   *
30   * <ul>
31   * <li>Obtain a object representation of a file header from a existing file
32   * (String to FileHeader).</li>
33   * <li>Obtain the file header to inject in a file from a FileHeader (to update
34   * or create a file header from the file header model).
35   * </ul>
36   *
37   * Moreover the contract offers two methods to box and unbox a String
38   * representation of a header content :
39   * <ul>
40   * <li>{@link #boxComment(String, boolean)}</li>
41   * <li>{@link #unboxComent(String)}</li>
42   * </ul>
43   *
44   * @author tchemit dev@tchemit.fr
45   * @since 1.0
46   */
47  public interface FileHeaderTransformer
48  {
49  
50      /**
51       * default section delimiter.
52       */
53      String DEFAULT_SECTION_DELIMITER = "%" + "%";
54  
55      /**
56       * default process start tag.
57       */
58      String DEFAULT_PROCESS_START_TAG = "#" + "%" + "L";
59  
60      /**
61       * default process end tag.
62       */
63      String DEFAULT_PROCESS_END_TAG = "#" + "L" + "%";
64  
65      /**
66       * Line separator.
67       */
68      String LINE_SEPARATOR = System.getProperty("line.separator");
69  
70      /**
71       * @return the name of the transformer
72       */
73      String getName();
74  
75      /**
76       * @return the description of the transformer
77       */
78      String getDescription();
79  
80      /**
81       * Get the default accepted extensions for this transformer.
82       *
83       * @return the default accepted extensions.
84       */
85      String[] getDefaultAcceptedExtensions();
86  
87      /**
88       * Obtains the process tag which indicates the begin of the header content.
89       *
90       * By default, (says if you do not explicitly invoke the
91       * {@link #setProcessStartTag(String)} method), will use the
92       * {@link #DEFAULT_PROCESS_START_TAG}
93       *
94       * @return the starting header tag
95       */
96      String getProcessStartTag();
97  
98      /**
99       * Obtain the process tag which indiciates the end of the header content.
100      *
101      * By default, (says if you do not explicitly invoke the
102      * {@link #setProcessEndTag(String)} method), will use the
103      * {@link #DEFAULT_PROCESS_END_TAG}.
104      *
105      * @return the ending header tag
106      */
107     String getProcessEndTag();
108 
109     /**
110      * The pattern used to separate sections of the header.
111      *
112      * By default, (says if you do not explicitly invoke the
113      * {@link #setSectionDelimiter(String)} method), will use the
114      * {@link #DEFAULT_SECTION_DELIMITER}.
115      *
116      * @return the delimiter used to separate sections in the header.
117      */
118     String getSectionDelimiter();
119 
120     /**
121      * @return the start tag of a comment
122      */
123     String getCommentStartTag();
124 
125     /**
126      * @return the end tag of a comment
127      */
128     String getCommentEndTag();
129 
130     /**
131      * @return the line prefix of every line insed the comment
132      */
133     String getCommentLinePrefix();
134 
135     /**
136      * Get flag if there should be an empty line after the header.
137      *
138      * @return if there should be an empty line after the header
139      */
140     boolean isEmptyLineAfterHeader();
141 
142     /**
143      * Adds the header.
144      *
145      * @param header  header to add
146      * @param content content of original file
147      * @return the new full file content beginning with header
148      */
149     String addHeader( String header, String content );
150 
151     /**
152      * Box the given {@code header} in a comment.
153      *
154      * @param header   the header content WITHOUT any comment boxing
155      * @param withTags flag to add start and end comment tags.
156      * @return the header content WITH comment boxing
157      */
158     String boxComment( String header, boolean withTags );
159 
160     /**
161      * Unbox the given boxed {@code boxedHeader} to obtain the header content.
162      *
163      * @param boxedHeader the boxed header
164      * @return the unboxed header.
165      */
166     String unboxComent( String boxedHeader );
167 
168     /**
169      * Box the given {@code header} between process tags.
170      *
171      * @param header the header content WITHOUT any comment boxing
172      * @return the header content boxed between process tags
173      * @see #getProcessStartTag()
174      * @see #getProcessEndTag()
175      */
176     String boxProcessTag( String header );
177 
178     /**
179      * Unbox the process tag on the given boxed {@code boxedHeader} to obtain
180      * the brute header content.
181      *
182      * @param boxedHeader the boxed header
183      * @return the brute header content.
184      * @see #getProcessStartTag()
185      * @see #getProcessEndTag()
186      */
187     String unboxProcessTag( String boxedHeader );
188 
189     /**
190      * Build a {@link FileHeader} from an UNBOXED header content.
191      *
192      * @param header unboxed header content
193      * @return The model of the header content
194      */
195     FileHeader toFileHeader( String header );
196 
197     /**
198      * Build a UNBOXED header content from the given {@code model}.
199      *
200      * @param model the model of the file header (can not be null)
201      * @return the UNBOXED header content
202      */
203     String toString( FileHeader model );
204 
205     /**
206      * Build a fully boxed header content from the given {@code model}.
207      *
208      * @param model the model of the file header (can not be null)
209      * @return the fully boxed header content
210      */
211     String toHeaderContent( FileHeader model );
212 
213     /**
214      * Tests if the description of the two models are equals.
215      *
216      * @param header1 the first header
217      * @param header2 the second header
218      * @return {@code true} if headers description are stricly the same
219      */
220     boolean isDescriptionEquals( FileHeader header1, FileHeader header2 );
221 
222     /**
223      * Tests if the copyright of the two models are equals.
224      *
225      * @param header1 the first header
226      * @param header2 the second header
227      * @return {@code true} if headers copyright are stricly the same
228      */
229     boolean isCopyrightEquals( FileHeader header1, FileHeader header2 );
230 
231     /**
232      * Tests if the license of the two models are equals.
233      *
234      * @param header1 the first header
235      * @param header2 the second header
236      * @return {@code true} if headers license are stricly the same (WITHOUT ANY space)
237      */
238     boolean isLicenseEquals( FileHeader header1, FileHeader header2 );
239 
240     /**
241      * Changes the name of the transformer.
242      *
243      * @param name the new name of the transformer
244      */
245     void setName( String name );
246 
247     /**
248      * Chages the description of the transformer.
249      *
250      * @param description the new description of the transformer
251      */
252     void setDescription( String description );
253 
254     /**
255      * Sets the header section delimiter.
256      *
257      * By default, will use the {@link #DEFAULT_SECTION_DELIMITER}.
258      *
259      * @param headerSectionDelimiter the new delimiter
260      */
261     void setSectionDelimiter( String headerSectionDelimiter );
262 
263     /**
264      * Changes the process start tag.
265      *
266      * @param tag the new start tag
267      */
268     void setProcessStartTag( String tag );
269 
270     /**
271      * Changes the process end tag.
272      *
273      * @param tag the new endtag
274      */
275     void setProcessEndTag( String tag );
276 
277     /**
278      * Changes the comment start tag.
279      *
280      * @param commentStartTag the new comment start tag
281      */
282     void setCommentStartTag( String commentStartTag );
283 
284     /**
285      * Changes the comment end tag.
286      *
287      * @param commentEndTag the new comment end tag
288      */
289     void setCommentEndTag( String commentEndTag );
290 
291     /**
292      * Changes the comment prefix line.
293      *
294      * @param commentLinePrefix the new comment prefix line
295      */
296     void setCommentLinePrefix( String commentLinePrefix );
297 
298     /**
299      * Set flag if there should be an empty line after the header.
300      *
301      * @param emptyLine flag if there should be an empty line after the header
302      */
303     void setEmptyLineAfterHeader( boolean emptyLine );
304 }
