25  import org.codehaus.plexus.component.annotations.Component;
26  
27  /**
28   * Implementation of {@link FileHeaderTransformer} for sql format.
29   *
30   * @author tchemit dev@tchemit.fr
31   * @since 1.0
32   */
33  @Component( role = FileHeaderTransformer.class, hint = "sql" )
34  public class SqlFileHeaderTransformer
35      extends AbstractFileHeaderTransformer
36  {
37  
38      /**
39       * Default constructor.
40       */
41      public SqlFileHeaderTransformer()
42      {
43          super( "sql", "header transformer with sql comment style", "---", "---", "-- " );
44      }
45  
46      /**
47       * {@inheritDoc}
48       */
49      public String[] getDefaultAcceptedExtensions()
50      {
51          return new String[]{ "sql" };
52      }
53  }
