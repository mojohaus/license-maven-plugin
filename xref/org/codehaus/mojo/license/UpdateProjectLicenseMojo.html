25  import org.apache.maven.plugins.annotations.LifecyclePhase;
26  import org.apache.maven.plugins.annotations.Mojo;
27  import org.apache.maven.plugins.annotations.Parameter;
28  import org.codehaus.mojo.license.model.License;
29  import org.codehaus.mojo.license.utils.FileUtil;
30  import org.slf4j.Logger;
31  import org.slf4j.LoggerFactory;
32  
33  import java.io.File;
34  
35  /**
36   * Updates (or creates) the main project license file according to the given
37   * license defines as {@link #licenseName}.
38   *
39   * Can also generate a bundled license file (to avoid collision names in
40   * class-path). This file is by default generated in
41   * {@code META-INF class-path directory}.
42   *
43   * @author tchemit dev@tchemit.fr
44   * @since 1.0
45   */
46  @Mojo( name = "update-project-license", defaultPhase = LifecyclePhase.GENERATE_RESOURCES )
47  public class UpdateProjectLicenseMojo
48      extends AbstractLicenseNameMojo
49  {
50      private static final Logger LOG = LoggerFactory.getLogger( UpdateProjectLicenseMojo.class );
51  
52      // ----------------------------------------------------------------------
53      // Mojo Parameters
54      // ----------------------------------------------------------------------
55  
56      /**
57       * Project license file to synchronize with main license defined in
58       * descriptor file.
59       *
60       * @required
61       * @since 1.0
62       */
63      @Parameter( property = "license.licenceFile", defaultValue = "${basedir}/LICENSE.txt" )
64      protected File licenseFile;
65  
66      /**
67       * The directory where to generate license resources.
68       *
69       * <b>Note:</b> This option is not available for {@code pom} module types.
70       *
71       * @since 1.0
72       */
73      @Parameter( property = "license.outputDirectory", defaultValue = "target/generated-sources/license" )
74      protected File outputDirectory;
75  
76      /**
77       * A flag to copy the main license file in a bundled place.
78       *
79       * This is usefull for final application to have a none confusing location
80       * to seek for the application license.
81       *
82       * If Sets to {@code true}, will copy the license file to the
83       * {@link #bundleLicensePath} to {@link #outputDirectory}.
84       *
85       * <b>Note:</b> This option is not available for {@code pom} module types.
86       *
87       * @since 1.0
88       */
89      @Parameter( property = "license.generateBundle", defaultValue = "false" )
90      protected boolean generateBundle;
91  
92      /**
93       * The path of the bundled license file to produce when
94       * {@link #generateBundle} is on.
95       *
96       * <b>Note:</b> This option is not available for {@code pom} module types.
97       *
98       * @since 1.0
99       */
100     @Parameter( property = "license.bundleLicensePath", defaultValue = "META-INF/${project.artifactId}-LICENSE.txt" )
101     protected String bundleLicensePath;
102 
103     /**
104      * A flag to force to generate project license file even if it is up-to-date.
105      *
106      * @since 1.0.0
107      */
108     @Parameter( property = "license.force", defaultValue = "false" )
109     protected boolean force;
110 
111     /**
112      * A flag to skip the goal.
113      *
114      * @since 1.0
115      */
116     @Parameter( property = "license.skipUpdateProjectLicense", defaultValue = "false" )
117     protected boolean skipUpdateProjectLicense;
118 
119     // ----------------------------------------------------------------------
120     // Private Fields
121     // ----------------------------------------------------------------------
122 
123     /**
124      * Flag to known if generate is needed.
125      */
126     private boolean doGenerate;
127 
128     // ----------------------------------------------------------------------
129     // AbstractLicenceMojo Implementation
130     // ----------------------------------------------------------------------
131 
132     /**
133      * {@inheritDoc}
134      */
135     @Override
136     public boolean isSkip()
137     {
138         return skipUpdateProjectLicense;
139     }
140 
141     /**
142      * {@inheritDoc}
143      */
144     @Override
145     protected void init()
146         throws Exception
147     {
148         super.init();
149 
150         // must generate if file does not exist or pom never thant license file
151         if ( licenseFile != null )
152         {
153             File pomFile = getProject().getFile();
154 
155             this.doGenerate = force || !licenseFile.exists() || licenseFile.lastModified() <= pomFile.lastModified();
156         }
157     }
158 
159     /**
160      * {@inheritDoc}
161      */
162     @Override
163     protected void doAction()
164         throws Exception
165     {
166 
167         License license = getLicense();
168 
169         if ( doGenerate )
170         {
171 
172             LOG.info( "Will create or update license file [{}] to {}", license.getName(), licenseFile );
173             if ( isVerbose() )
174             {
175                 LOG.info( "detail of license :\n{}", license );
176             }
177 
178             if ( licenseFile.exists() && isKeepBackup() )
179             {
180                 if ( isVerbose() )
181                 {
182                     LOG.info( "backup {}", licenseFile );
183                 }
184                 // copy it to backup file
185                 FileUtil.backupFile( licenseFile );
186             }
187         }
188 
189         // obtain license content
190         String licenseContent = license.getLicenseContent( getEncoding() );
191 
192         if ( license.isLicenseContentTemplateAware() )
193         {
194 
195             licenseContent = processLicenseContext( licenseContent );
196         }
197         if ( doGenerate )
198         {
199 
200             // writes it root main license file
201             FileUtil.printString( licenseFile, licenseContent, getEncoding() );
202         }
203 
204         if ( hasClassPath() )
205         {
206 
207             // copy the license file to the resources directory
208             File resourceTarget = new File( outputDirectory, licenseFile.getName() );
209             FileUtil.copyFile( this.licenseFile, resourceTarget );
210 
211             addResourceDir( outputDirectory, "**/" + resourceTarget.getName() );
212 
213             if ( generateBundle )
214             {
215 
216                 // creates the bundled license file
217                 File bundleTarget = FileUtil.getFile( outputDirectory, bundleLicensePath );
218                 FileUtil.copyFile( licenseFile, bundleTarget );
219 
220                 if ( !resourceTarget.getName().equals( bundleTarget.getName() ) )
221                 {
222 
223                     addResourceDir( outputDirectory, "**/" + bundleTarget.getName() );
224                 }
225             }
226         }
227     }
228 }
