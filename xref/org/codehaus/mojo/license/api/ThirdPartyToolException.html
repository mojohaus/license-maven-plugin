25  /**
26   * An exception occurring during the execution of this tool.
27   *
28   * @author <a href="mailto:tchemit@codelutin.com">tony chemit</a>
29   * @version $Id$
30   */
31  public class ThirdPartyToolException
32      extends Exception
33  {
34      /**
35       * Construct a new <code>ThirdPartyToolException</code> exception wrapping an underlying <code>Exception</code>
36       * and providing a <code>message</code>.
37       *
38       * @param message could be null
39       * @param cause   could be null
40       */
41      public ThirdPartyToolException( String message, Exception cause )
42      {
43          super( message, cause );
44      }
45  
46      /**
47       * Construct a new <code>ThirdPartyToolException</code> exception wrapping an underlying <code>Throwable</code>
48       * and providing a <code>message</code>.
49       *
50       * @param message could be null
51       * @param cause   could be null
52       */
53      public ThirdPartyToolException( String message, Throwable cause )
54      {
55          super( message, cause );
56      }
57  
58      /**
59       * Construct a new <code>ThirdPartyToolException</code> exception providing a <code>message</code>.
60       *
61       * @param message could be null
62       */
63      public ThirdPartyToolException( String message )
64      {
65          super( message );
66      }
67  }
