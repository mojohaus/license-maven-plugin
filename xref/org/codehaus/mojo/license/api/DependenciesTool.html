25  import org.apache.maven.artifact.repository.ArtifactRepository;
26  import org.apache.maven.project.MavenProject;
27  
28  import java.util.List;
29  import java.util.SortedMap;
30  
31  /**
32   * A tool to deal with dependencies of a project.
33   *
34   * @author tchemit dev@tchemit.fr
35   * @since 1.0
36   */
37  public interface DependenciesTool
38  {
39  
40      /**
41       * For a given {@code project}, obtain the universe of its dependencies after applying transitivity and
42       * filtering rules given in the {@code configuration} object.
43       *
44       * Result is given in a map where keys are unique artifact id
45       *
46       * @param project            the project to scann
47       * @param configuration      the configuration
48       * @param localRepository    local repository used to resolv dependencies
49       * @param remoteRepositories remote repositories used to resolv dependencies
50       * @param cache              a optional cache where to keep resolved dependencies
51       * @return the map of resolved dependencies indexed by their unique id.
52       * @see MavenProjectDependenciesConfigurator
53       */
54      SortedMap<String, MavenProject> loadProjectDependencies( MavenProject project,
55                                                               MavenProjectDependenciesConfigurator configuration,
56                                                               ArtifactRepository localRepository,
57                                                               List<ArtifactRepository> remoteRepositories,
58                                                               SortedMap<String, MavenProject> cache );
59  
60      /**
61       * Load project artifacts.
62       *
63       * @param project            the project to scann
64       * @param localRepository    local repository used to resolv dependencies
65       * @param remoteRepositories remote repositories used to resolv dependencies
66       * @throws DependenciesToolException if could not load project dependencies
67       */
68      void loadProjectArtifacts( ArtifactRepository localRepository, List remoteRepositories, MavenProject project )
69          throws DependenciesToolException;
70  }
