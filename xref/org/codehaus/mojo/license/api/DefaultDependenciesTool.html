100     public SortedMap<String, MavenProject> loadProjectDependencies( ResolvedProjectDependencies artifacts,
101                                                                     MavenProjectDependenciesConfigurator configuration,
102                                                                     ArtifactRepository localRepository,
103                                                                     List<ArtifactRepository> remoteRepositories,
104                                                                     SortedMap<String, MavenProject> cache )
105     {
106 
107         boolean haveNoIncludedGroups = StringUtils.isEmpty( configuration.getIncludedGroups() );
108         boolean haveNoIncludedArtifacts = StringUtils.isEmpty( configuration.getIncludedArtifacts() );
109         boolean excludeTransitiveDependencies = configuration.isExcludeTransitiveDependencies();
110 
111         boolean haveExcludedGroups = StringUtils.isNotEmpty( configuration.getExcludedGroups() );
112         boolean haveExcludedArtifacts = StringUtils.isNotEmpty( configuration.getExcludedArtifacts() );
113         boolean haveExclusions = haveExcludedGroups || haveExcludedArtifacts;
114 
115         Pattern includedGroupPattern = null;
116         Pattern includedArtifactPattern = null;
117         Pattern excludedGroupPattern = null;
118         Pattern excludedArtifactPattern = null;
119 
120         if ( !haveNoIncludedGroups )
121         {
122             includedGroupPattern = Pattern.compile( configuration.getIncludedGroups() );
123         }
124         if ( !haveNoIncludedArtifacts )
125         {
126             includedArtifactPattern = Pattern.compile( configuration.getIncludedArtifacts() );
127         }
128         if ( haveExcludedGroups )
129         {
130             excludedGroupPattern = Pattern.compile( configuration.getExcludedGroups() );
131         }
132         if ( haveExcludedArtifacts )
133         {
134             excludedArtifactPattern = Pattern.compile( configuration.getExcludedArtifacts() );
135         }
136 
137         Set<?> depArtifacts;
138 
139         if ( configuration.isIncludeTransitiveDependencies() )
140         {
141             // All project dependencies
142             depArtifacts = artifacts.getAllDependencies();
143         }
144         else
145         {
146             // Only direct project dependencies
147             depArtifacts = artifacts.getDirectDependencies();
148         }
149 
150         List<String> includedScopes = configuration.getIncludedScopes();
151         List<String> excludeScopes = configuration.getExcludedScopes();
152         List<String> includedTypes = configuration.getIncludedTypes();
153         List<String> excludeTypes = configuration.getExcludedTypes();
154 
155         boolean verbose = configuration.isVerbose();
156 
157         SortedMap<String, MavenProject> result = new TreeMap<>();
158 
159         Map<String, Artifact> excludeArtifacts = new HashMap<>();
160         Map<String, Artifact> includeArtifacts = new HashMap<>();
161 
162         SortedMap<String, MavenProject> localCache = new TreeMap<>();
163         if ( cache != null )
164         {
165             localCache.putAll( cache );
166         }
167 
168         for ( Object o : depArtifacts )
169         {
170             Artifact artifact = (Artifact) o;
171 
172             excludeArtifacts.put( artifact.getId(), artifact );
173 
174             if ( DefaultThirdPartyTool.LICENSE_DB_TYPE.equals( artifact.getType() ) )
175             {
176                 // the special dependencies for license databases don't count.
177                 // Note that this will still see transitive deps of a license db; so using the build helper inside of
178                 // another project to make them will be noisy.
179                 continue;
180             }
181 
182             String scope = artifact.getScope();
183             if ( CollectionUtils.isNotEmpty( includedScopes ) && !includedScopes.contains( scope ) )
184             {
185 
186                 // not in included scopes
187                 continue;
188             }
189 
190             if ( excludeScopes.contains( scope ) )
191             {
192 
193                 // in excluded scopes
194                 continue;
195             }
196 
197             String type = artifact.getType();
198             if ( CollectionUtils.isNotEmpty( includedTypes ) && !includedTypes.contains( type ) )
199             {
200 
201                 // not in included scopes
202                 continue;
203             }
204 
205             if ( excludeTypes.contains( type ) )
206             {
207 
208                 // in excluded types
209                 continue;
210             }
211 
212             Logger log = getLogger();
213 
214             String id = MojoHelper.getArtifactId( artifact );
215 
216             if ( verbose )
217             {
218                 log.info( "detected artifact " + id );
219             }
220 
221             // Check if the project should be included
222             // If there is no specified artifacts and group to include, include all
223             boolean isToInclude = haveNoIncludedArtifacts && haveNoIncludedGroups
224                     || isIncludable( artifact, includedGroupPattern, includedArtifactPattern );
225 
226             // Check if the project should be excluded
227             boolean isToExclude = isToInclude && haveExclusions
228                     && isExcludable( artifact, excludedGroupPattern, excludedArtifactPattern );
229 
230             if ( !isToInclude || isToExclude )
231             {
232                 if ( verbose )
233                 {
234                     log.info( "skip artifact " + id );
235                 }
236                 continue;
237             }
238 
239             MavenProject depMavenProject;
240 
241             // try to get project from cache
242             depMavenProject = localCache.get( id );
243 
244             if ( depMavenProject != null )
245             {
246                 if ( verbose )
247                 {
248                     log.info( "add dependency [" + id + "] (from cache)" );
249                 }
250             }
251             else
252             {
253                 // build project
254 
255                 try
256                 {
257                     depMavenProject =
258                         mavenProjectBuilder.buildFromRepository( artifact, remoteRepositories, localRepository, true );
259                     depMavenProject.getArtifact().setScope( artifact.getScope() );
260 
261                     // In case maven-metadata.xml has different artifactId, groupId or version.
262                     if ( !depMavenProject.getGroupId().equals( artifact.getGroupId() ) )
263                     {
264                         depMavenProject.setGroupId( artifact.getGroupId() );
265                         depMavenProject.getArtifact().setGroupId( artifact.getGroupId() );
266                     }
267                     if ( !depMavenProject.getArtifactId().equals( artifact.getArtifactId() ) )
268                     {
269                         depMavenProject.setArtifactId( artifact.getArtifactId() );
270                         depMavenProject.getArtifact().setArtifactId( artifact.getArtifactId() );
271                     }
272                     if ( !depMavenProject.getVersion().equals( artifact.getVersion() ) )
273                     {
274                         depMavenProject.setVersion( artifact.getVersion() );
275                         depMavenProject.getArtifact().setVersion( artifact.getVersion() );
276                     }
277                 }
278                 catch ( ProjectBuildingException e )
279                 {
280                     log.warn( "Unable to obtain POM for artifact : " + artifact, e );
281                     continue;
282                 }
283 
284                 if ( verbose )
285                 {
286                     log.info( "add dependency [" + id + "]" );
287                 }
288 
289                 // store it also in cache
290                 localCache.put( id, depMavenProject );
291             }
292 
293             // keep the project
294             result.put( id, depMavenProject );
295 
296             excludeArtifacts.remove( artifact.getId() );
297             includeArtifacts.put( artifact.getId(), artifact );
298         }
299 
300         // exclude artifacts from the result that contain excluded artifacts in the dependency trail
301         if ( excludeTransitiveDependencies )
302         {
303             for ( Map.Entry<String, Artifact> entry : includeArtifacts.entrySet() )
304             {
305                 List<String> dependencyTrail = entry.getValue().getDependencyTrail();
306 
307                 boolean remove = false;
308 
309                 for ( int i = 1; i < dependencyTrail.size() - 1; i++ )
310                 {
311                     if ( excludeArtifacts.containsKey( dependencyTrail.get( i ) ) )
312                     {
313                         remove = true;
314                         break;
315                     }
316                 }
317 
318                 if ( remove )
319                 {
320                     result.remove( MojoHelper.getArtifactId( entry.getValue() ) );
321                 }
322             }
323         }
324 
325         if ( cache != null )
326         {
327             cache.putAll( result );
328         }
329 
330         return result;
331     }

334     /**
335      * {@inheritDoc}
336      */
337     public ResolvedProjectDependencies loadProjectArtifacts( ArtifactRepository localRepository,
338             List remoteRepositories, MavenProject project, List<MavenProject> reactorProjects )
339         throws DependenciesToolException
340 
341     {
342         Map<String, MavenProject> idToReactorProject = new HashMap<>();
343         if ( reactorProjects != null )
344         {
345             for ( MavenProject reactorProject : reactorProjects )
346             {
347                 idToReactorProject.put( String.format( "%s:%s", reactorProject.getGroupId(),
348                         reactorProject.getArtifactId() ), reactorProject );
349             }
350         }
351 
352         /*
353          * Find the list of dependencies to resolve transitively. Some projects may be in the reactor.
354          * Reactor projects can't be resolved by the artifact resolver yet.
355          * In order to still get the complete dependency tree for the project, we will add the transitive
356          * dependencies of the reactor project to the list of dependencies to resolve.
357          * Since the transitive dependencies could
358          * also be reactor projects, we need to repeat this check for each of those.
359          * Note that since the dependency reactor
360          * project may specify its own list of repositories,
361          * we need to keep track of which project the transitive dependency is declared in.
362          */
363         List<Dependency> directDependencies = new ArrayList<>( project.getDependencies() );
364         Map<MavenProject, List<Dependency>> reactorProjectToTransitiveDependencies = new HashMap<>();
365         Queue<Pair<MavenProject, Dependency>> dependenciesToCheck = new ArrayDeque<>();
366         for ( Dependency dependency : directDependencies )
367         {
368             dependenciesToCheck.add( Pair.of( project, dependency ) );
369         }
370         if ( reactorProjects != null )
371         {
372             while ( !dependenciesToCheck.isEmpty() )
373             {
374                 Pair<MavenProject, Dependency> pair = dependenciesToCheck.remove();
375                 Dependency dependency = pair.getRight();
376                 String id = String.format( "%s:%s", dependency.getGroupId(), dependency.getArtifactId() );
377                 MavenProject dependencyReactorProject = idToReactorProject.get( id );
378                 if ( dependencyReactorProject != null )
379                 {
380                     /*
381                      * Since the project is in the reactor, the artifact resolver may not be able to resolve
382                      * the artifact plus transitive dependencies yet. In order to still get the
383                      * complete dependency tree for the project, we will add the transitive
384                      * dependencies of the reactor project to the list of dependencies to resolve.
385                      * Since the transitive dependencies could
386                      * also be reactor projects, we need to repeat this check for each of those.
387                      * Note that since the dependency reactor
388                      * project may specify its own list of repositories,
389                      * we need to keep track of which project the transitive dependency is
390                      * declared in.
391                      */
392                     for ( Dependency transitiveDependency
393                             : ( List<Dependency> ) dependencyReactorProject.getDependencies() )
394                     {
395                         dependenciesToCheck.add( Pair.of( dependencyReactorProject, transitiveDependency ) );
396                     }
397                 }
398                 if ( !directDependencies.contains( dependency ) )
399                 {
400                     List<Dependency> transitiveForSameProject =
401                             reactorProjectToTransitiveDependencies.get( pair.getLeft() );
402                     if ( transitiveForSameProject == null )
403                     {
404                         transitiveForSameProject = new ArrayList<>();
405                         reactorProjectToTransitiveDependencies.put( pair.getLeft(), transitiveForSameProject );
406                     }
407                     transitiveForSameProject.add( dependency );
408                 }
409             }
410         }
411 
412         //Create artifacts for all dependencies,
413         //keep the transitive dependencies grouped by project they are declared in
414         Set<Artifact> directDependencyArtifacts = createDependencyArtifacts( project, directDependencies );
415         Map<MavenProject, Set<Artifact>> reactorProjectToDependencyArtifacts = new HashMap<>();
416         for ( Entry<MavenProject, List<Dependency>> entry : reactorProjectToTransitiveDependencies.entrySet() )
417         {
418             reactorProjectToDependencyArtifacts.put( entry.getKey(),
419                     createDependencyArtifacts( entry.getKey(), entry.getValue() ) );
420         }
421 
422         //Resolve artifacts. Transitive dependencies are resolved with the settings of the POM they are declared in.
423         //Skip reactor projects, since they can't necessarily be resolved yet.
424         //The transitive handling above ensures we still get a complete list of dependencies.
425         Set<Artifact> reactorArtifacts = new HashSet<>();
426         Set<Artifact> directArtifactsToResolve = new HashSet<>();
427         if ( reactorProjects == null )
428         {
429             directArtifactsToResolve.addAll( directDependencyArtifacts );
430         }
431         else
432         {
433             partitionByIsReactorProject( directDependencyArtifacts, reactorArtifacts,
434                     directArtifactsToResolve, idToReactorProject.keySet() );
435             for ( Entry<MavenProject, Set<Artifact>> entry : reactorProjectToDependencyArtifacts.entrySet() )
436             {
437                 Set<Artifact> nonReactorArtifacts = new HashSet<>();
438                 partitionByIsReactorProject( entry.getValue(), reactorArtifacts,
439                         nonReactorArtifacts, idToReactorProject.keySet() );
440                 entry.setValue( nonReactorArtifacts );
441             }
442         }
443         Set<Artifact> allDependencies = new HashSet<>( reactorArtifacts );
444         allDependencies.addAll( resolve( directArtifactsToResolve, project.getArtifact(), localRepository,
445                 remoteRepositories ).getArtifacts() );
446         for ( Entry<MavenProject, Set<Artifact>> entry : reactorProjectToDependencyArtifacts.entrySet() )
447         {
448             MavenProject reactorProject = entry.getKey();
449             Set<Artifact> toResolve = entry.getValue();
450             Artifact reactorProjectArtifact = reactorProject.getArtifact();
451             List<ArtifactRepository> reactorRemoteRepositories = reactorProject.getRemoteArtifactRepositories();
452             allDependencies.addAll(
453                     resolve( toResolve, reactorProjectArtifact, localRepository,
454                             reactorRemoteRepositories ).getArtifacts() );
455         }
456 
457         return new ResolvedProjectDependencies( allDependencies, directDependencyArtifacts );
458     }

460     private Set<Artifact> createDependencyArtifacts( MavenProject project, List<Dependency> dependencies )
461             throws DependenciesToolException
462     {
463         try
464         {
465             return MavenMetadataSource.createArtifacts( artifactFactory, dependencies, null, null, project );
466         }
467         catch ( InvalidDependencyVersionException e )
468         {
469             throw new DependenciesToolException( e );
470         }
471     }
472 
473     private void partitionByIsReactorProject( Set<Artifact> artifacts, Set<Artifact> reactorArtifacts,
474             Set<Artifact> nonReactorArtifacts, Set<String> reactorProjectIds )
475     {
476         for ( Artifact dependencyArtifact : artifacts )
477         {
478                 String artifactKey = String.format( "%s:%s", dependencyArtifact.getGroupId(),
479                         dependencyArtifact.getArtifactId() );
480                 if ( reactorProjectIds.contains( artifactKey ) )
481                 {
482                     reactorArtifacts.add( dependencyArtifact );
483                 }
484                 else
485                 {
486                     nonReactorArtifacts.add( dependencyArtifact );
487                 }
488             }
489     }

491     private ArtifactResolutionResult resolve( Set<Artifact> artifacts, Artifact projectArtifact,
492             ArtifactRepository localRepository, List remoteRepositories )
493         throws DependenciesToolException
494     {
495         try
496         {
497             return artifactResolver.resolveTransitively( artifacts, projectArtifact, remoteRepositories,
498                                                            localRepository, artifactMetadataSource );
499         }
500         catch ( ArtifactResolutionException e )
501         {
502             throw new DependenciesToolException( e );
503         }
504         catch ( ArtifactNotFoundException e )
505         {
506             throw new DependenciesToolException( e );
507             }
508     }

510     /**
511      * Tests if the given project is includeable against a groupdId pattern and a artifact pattern.
512      *
513      * @param project                 the project to test
514      * @param includedGroupPattern    the include group pattern
515      * @param includedArtifactPattern the include artifact pattenr
516      * @return {@code true} if the project is includavble, {@code false} otherwise
517      */
518     protected boolean isIncludable( Artifact project, Pattern includedGroupPattern, Pattern includedArtifactPattern )
519     {
520 
521         Logger log = getLogger();
522 
523         // check if the groupId of the project should be included
524         if ( includedGroupPattern != null )
525         {
526             // we have some defined license filters
527             try
528             {
529                 Matcher matchGroupId = includedGroupPattern.matcher( project.getGroupId() );
530                 if ( matchGroupId.find() )
531                 {
532                     if ( log.isDebugEnabled() )
533                     {
534                         log.debug( "Include " + project.getGroupId() );
535                     }
536                     return true;
537                 }
538             }
539             catch ( PatternSyntaxException e )
540             {
541                 log.warn( String.format( INVALID_PATTERN_MESSAGE, includedGroupPattern.pattern() ) );
542             }
543         }
544 
545         // check if the artifactId of the project should be included
546         if ( includedArtifactPattern != null )
547         {
548             // we have some defined license filters
549             try
550             {
551                 Matcher matchGroupId = includedArtifactPattern.matcher( project.getArtifactId() );
552                 if ( matchGroupId.find() )
553                 {
554                     if ( log.isDebugEnabled() )
555                     {
556                         log.debug( "Include " + project.getArtifactId() );
557                     }
558                     return true;
559                 }
560             }
561             catch ( PatternSyntaxException e )
562             {
563                 log.warn( String.format( INVALID_PATTERN_MESSAGE, includedArtifactPattern.pattern() ) );
564             }
565         }
566         return false;
567     }

578     {
579 
580         Logger log = getLogger();

594                     }
595                     return true;
596                 }
597             }
598             catch ( PatternSyntaxException e )
599             {
600                 log.warn( String.format( INVALID_PATTERN_MESSAGE, excludedGroupPattern.pattern() ) );
601             }
602         }
603 
604         // check if the artifactId of the project should be included
605         if ( excludedArtifactPattern != null )

622                 log.warn( String.format( INVALID_PATTERN_MESSAGE, excludedArtifactPattern.pattern() ) );

624         }
