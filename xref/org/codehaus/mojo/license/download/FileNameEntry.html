25  import java.io.File;
26  
27  /**
28   * A triple consiting of a {@link File}, its SHA-1 checksum and a boolean whether {@link #file}'s name comes from
29   * {@link PreferredFileNames} and is thus preferred.
30   *
31   * @author <a href="https://github.com/ppalaga">Peter Palaga</a>
32   * @since 1.18
33   */
34  public class FileNameEntry
35  {
36      public FileNameEntry( File file, boolean preferred, String sha1 )
37      {
38          super();
39          this.file = file;
40          this.preferred = preferred;
41          this.sha1 = sha1;
42      }
43  
44      private final File file;
45  
46      private final boolean preferred;
47  
48      private final String sha1;
49  
50      public File getFile()
51      {
52          return file;
53      }
54  
55      public boolean isPreferred()
56      {
57          return preferred;
58      }
59  
60      public String getSha1()
61      {
62          return sha1;
63      }
64  }
