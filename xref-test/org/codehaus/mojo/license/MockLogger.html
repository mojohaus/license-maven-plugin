3   import java.util.regex.Matcher;
4   import java.util.regex.Pattern;
5   
6   import org.codehaus.mojo.license.LicenseMojoUtils.LoggerFacade;
7   
8   public class MockLogger
9       implements LoggerFacade
10  {
11      private static final Pattern PLACEHOLDER_PATTERN = Pattern.compile( "\\Q{}\\E" );
12  
13      private static final String format( String template, Object param )
14      {
15          return PLACEHOLDER_PATTERN.matcher( template )
16                          .replaceFirst( Matcher.quoteReplacement( String.valueOf( param ) ) );
17      }
18  
19      private final StringBuilder log = new StringBuilder();
20  
21      @Override
22      public void debug( String template, Object param )
23      {
24          log.append( "DEBUG " ).append( format( template, param ) ).append( "\n" );
25      }
26  
27      @Override
28      public void warn( String string )
29      {
30          log.append( "WARN " ).append( string ).append( "\n" );
31      }
32  
33      @Override
34      public void warn( String template, Object param )
35      {
36          log.append( "WARN " ).append( format( template, param ) ).append( "\n" );
37      }
38  
39      public String dump()
40      {
41          return log.toString();
42      }
43  
44      public void reset()
45      {
46          log.setLength( 0 );
47      }
48  
49  }
